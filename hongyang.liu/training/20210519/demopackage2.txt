        def main(args:Array[String]):Unit={

            var dog = new Dog
            println("dog="+dog)

            //在子包和父包类重名时，默认采用就近原则
            var u = new User()
            println("u"+u)
            //在子包和父包类重名时，如果希望指定使用某个类，则带上包路径
